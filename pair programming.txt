#include <bits/stdc++.h>
using namespace std;
 
 
 
int main()
{
     
#ifndef ONLINE_JUDGE
        freopen("intant.txt","r",stdin);
        freopen("outant.txt","w",stdout);
        
    #endif
        
     int t;
     cin>>t;
     while(t--)
     {
          int k,m,n,cnt=0,x=0,i,pr,ni,out=0,mi,sr;
          cin>>k>>m>>n;
          int a[m];
          int b[n];
          int c[m+n];
          for(i=0;i<m;i++)
            {
               cin>>a[i];
               if(a[i]>0)
                c[x++]=a[i];
              if(a[i]==0)
                cnt++;
         
            }
           for(i=0;i<n;i++)
            {
              cin>>b[i];
              if(b[i]>0)
                c[x++]=b[i];
              if(b[i]==0)
                cnt++;

            }
            pr=x;
            sr=k;
         
          ni=cnt;
          mi=cnt;
            for(i=0;i<x;i++)
            {
              if(c[i]-sr<=mi)
              {
                
                mi=mi-c[i]+sr;
                sr=c[i];
              }
              else
              {
                out=1;
              }
            }
         if(out)
       cout<<"-1";

          if(!out)
        {  
         
            
          for(i=0;i<m+n-ni;i++)
          {
             if(c[i]<=k)
              { cout<<c[i]<<" ";
                  
                
              }
              else if(c[i]>k )
              {
                 while(c[i]>k && cnt>0)
                 {
                   cout<<"0"<<" ";
                   k++;
                   cnt--;

                 }
                 if(c[i]==k)
                 {
                  cout<<c[i]<<" ";
                 }
                 
                

              }

          }


      }
      cout<<endl;
     }
    
     
 }
